        function navigate(pageId) {

                l.classList.add('text-gray-700', 'dark:text-gray-300');

            if(window.innerWidth < 768) {
                document.getElementById('sidebar').classList.add('-translate-x-full');
                document.getElementById('sidebar-overlay').classList.add('hidden');
            }
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }

        function speakElementContent(elementId) {
            if (!window.speechSynthesis) return;
            window.speechSynthesis.cancel();
            const element = document.getElementById(elementId);
            if (!element) return;
            let tempDiv = document.createElement('div');
            tempDiv.innerHTML = element.innerHTML.replace(/<br\s*[\/]?>/gi, " ");
            let textToSpeak = (tempDiv.innerText || tempDiv.textContent).replace(/\s+/g, ' ').trim();
            playTTS(textToSpeak);
        }

        function speakCurrentPage() {
            if (!window.speechSynthesis) return;
            if (window.speechSynthesis.speaking) { window.speechSynthesis.cancel(); return; }
            
            const modal = document.getElementById('global-modal');
            if(!modal.classList.contains('hidden')) {
                speakElementContent('modal-body');
                return;
            }

            const activeSection = document.querySelector('.page-section:not(.hidden)');
            if (!activeSection) return;
            if (activeSection.id === 'home') {
                playTTS(translations[currentLang]['hero_title'] + "。" + translations[currentLang]['hero_subtitle']);
            } else {
                const contentWrapper = activeSection.querySelector('[id^="content-"]');
                if (contentWrapper) speakElementContent(contentWrapper.id);
            }
        }

        function playTTS(text) {
            const utterance = new SpeechSynthesisUtterance(text);
            const voices = window.speechSynthesis.getVoices();
            if (currentLang === 'en') {
                let v = voices.find(v => v.lang === 'en-GB' && (v.name.toLowerCase().includes('male') || v.name.includes('Daniel') || v.name.includes('Google UK English Male')));
                utterance.voice = v || voices.find(v => v.lang === 'en-GB');
            } else {
                let v = voices.find(v => v.lang === 'zh-TW' && v.name.toLowerCase().includes('yating'));
                utterance.voice = v || voices.find(v => v.lang === 'zh-TW' || v.lang === 'zh-HK');
            }
            utterance.rate = 0.9;
            window.speechSynthesis.speak(utterance);
        }

        // --- 互動測驗邏輯 (隨機抽題) ---
        function shuffleAndShowQuiz() {
            answeredQuestions.clear();
            const resultBox = document.getElementById('quiz-result');
            if(resultBox) resultBox.classList.add('hidden');
            
            const pool = document.getElementById('quiz-pool');
            if(!pool) return;
            
            const items = Array.from(pool.querySelectorAll('.quiz-item'));
            
            // 重置所有題目的狀態與樣式
            items.forEach(item => {
                item.classList.add('hidden');
                const btns = item.querySelectorAll('.quiz-btn');
                btns.forEach(b => {
                    b.disabled = false;
                    b.className = "quiz-btn w-full text-left px-6 py-4 rounded-xl border-2 border-gray-200 dark:border-gray-600 hover:bg-gray-50 dark:hover:bg-gray-700 transition-all font-medium text-lg";
                });
            });
            
            // 洗牌演算法 (Fisher-Yates Shuffle)
            for (let i = items.length - 1; i > 0; i--) {
                const j = Math.floor(Math.random() * (i + 1));
                [items[i], items[j]] = [items[j], items[i]];
            }
            
            // 放回 DOM 並顯示隨機前 5 題
            items.forEach((item, index) => {
                pool.appendChild(item);
                if(index < totalVisibleQuiz) {
                    item.classList.remove('hidden');
                }
            });
        }

        function answerQuiz(qId, isCorrect, btn) {
            if (answeredQuestions.has(qId)) return; // 已經答過該題
            answeredQuestions.add(qId);

            const container = btn.closest('.quiz-item');
            const buttons = container.querySelectorAll('.quiz-btn');
            
            // 禁用該題所有按鈕
            buttons.forEach(b => { 
                b.disabled = true; 
                b.classList.add('opacity-50', 'cursor-not-allowed'); 
                b.classList.remove('hover:bg-gray-50', 'dark:hover:bg-gray-700'); 
            });

            // 標示所選答案對錯
            btn.classList.remove('opacity-50');
            if (isCorrect) {
                btn.classList.add('bg-green-100', 'border-green-500', 'text-green-800', 'dark:bg-green-900', 'dark:border-green-400', 'dark:text-green-100');
            } else {
                btn.classList.add('bg-red-100', 'border-red-500', 'text-red-800', 'dark:bg-red-900', 'dark:border-red-400', 'dark:text-red-100');
            }

            // 若目前畫面上顯示的題目 (5題) 都答完了，則顯示結果
            if (answeredQuestions.size === totalVisibleQuiz) {
                const res = document.getElementById('quiz-result');
                res.textContent = translations[currentLang]['quiz_result_msg'];
                res.classList.remove('hidden');
            }
        }
